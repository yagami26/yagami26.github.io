Day 2 of learning golang with leetcode

I managed to solve 3 simple implementation problems today, all 3 related to string manipulation of some kind. I used the most brute force way to solve the problems, however I should check out the leetcode pattern for string problems. The thing is, I didn't want to go too quickly into the solution pattern as I am learning the implementation and I'm sure such problems will be arriving soon.

Anyway here's the code for all the problems and the links to the problem, so that I can review the syntax used.

Problem1

func strStr(haystack string, needle string) int {

    for i := 0; i < len(haystack); i++ {
        if haystack[i] == needle[0] {
            if isSubstring(i, haystack, needle) == true {
                return i
            }
        }
    }

    return -1
}

func isSubstring(startIdx int, haystack string, needle string) bool {
    i := 0
    for startIdx < len(haystack) && i < len(needle) && haystack[startIdx] == needle[i] {
        i = i + 1
        startIdx = startIdx + 1
    }

    if i == len(needle) {
        return true
    }

    return false
}

Problem2

func isAnagram(s string, t string) bool {
    smap, tmap := make(map[byte]int), make(map[byte]int)

    for i:= 0; i < len(s); i++ {
        smap[s[i]]++
    }

    for i:= 0; i < len(t); i++ {
        tmap[t[i]]++
    }


    for key, val := range smap {
        if tmap[key] != val {
            return false
        }
    }

    for key, val := range tmap {
        if smap[key] != val {
            return false
        }
    }

    return true
}

Problem3

func repeatedSubstringPattern(s string) bool {
    for chainLength:= 1; chainLength <= len(s)/2; chainLength++ {
        if doesChainRepeat(chainLength, s) {
            fmt.Println("Found chain of length ", chainLength)
            return true
        }
    }

    return false
}

func doesChainRepeat(chainLength int, s string) bool {
    for startIdxOne:= chainLength; startIdxOne < len(s); startIdxOne = startIdxOne + chainLength {
        fmt.Printf("Chain Length: %d, startIdxOne: %d\n", chainLength, startIdxOne)
        i,j := 0, startIdxOne
        for ; i < chainLength && j < len(s); i,j = i+1, j+1 {
            if (s[i] != s[j]) {
                return false
            }
        }

        if (i != chainLength) {
            return false
        }
    }

    return true
}
